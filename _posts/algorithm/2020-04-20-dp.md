## LIS

- 가장 긴 증가하는 부분 수열

~~~cpp
#include <iostream>
#include <algorithm>
using namespace std;

int dp[1001];
int arr[1001];
int ans = 0;


int main() {
	ios::sync_with_stdio(false); cin.tie(0); cout.tie(0);
	int n; cin >> n;

	for (int i{ 1 }; i <= n; i++) {
		cin >> arr[i];
		int prem = 0;
    //previous max value
		for (int prev{ 1 }; prev < i; prev++) {
      //index i 전까지의 arr[i] 보다 작은 부분 수열을 탐색
			if (arr[prev] < arr[i]) {
				prem = max(dp[prev], prem);
        //arr[i] 보다 작은 최대의 부분 수열
			}
		}
		dp[i] = prem + arr[i];
    //arr[i] 보다 작은 최대의 부분 수열의 개수 + 자신
		ans = max(ans, dp[i]);

    //dp array의 값들 중 최댓값
	}
	cout << ans;
	return 0;
}
~~~
